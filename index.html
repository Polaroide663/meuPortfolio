Exemplo de Código Python

Um trecho de código limpo e funcional que demonstra boas práticas de programação. 

# Script para análise de logs de erro
import re
from datetime import datetime

def contar_erros_por_data(arquivo_log: str, data_alvo: str) -> int:
    """
    Conta quantos erros ocorreram em uma data específica.
    
    Args:
        arquivo_log (str): Caminho do arquivo de log
        data_alvo (str): Data no formato 'YYYY-MM-DD'
    
    Returns:
        int: Número de erros encontrados
    """
    try:
        with open(arquivo_log, 'r', encoding='utf-8') as f:
            linhas = f.readlines()
        
        padrao = rf"^{data_alvo}\s.*ERROR"
        erros = [linha for linha in linhas if re.search(padrao, linha)]
        return len(erros)
    
    except FileNotFoundError:
        print("Arquivo de log não encontrado.")
        return 0

# Exemplo de uso
if __name__ == "__main__":
    data = datetime.today().strftime('%Y-%m-%d')
    total = contar_erros_por_data('app.log', data)
    print(f"Erros hoje ({data}): {total}")

Este script lê um arquivo de log e conta ocorrências de erros em uma data específica — útil para monitoramento de sistemas.
